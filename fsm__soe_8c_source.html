215 int ec_fsm_soe_prepare_read(
216  ec_fsm_soe_t *fsm,
217  ec_datagram_t *datagram
218  )
219 {
220  uint8_t *data;
221  ec_slave_t *slave = fsm->slave;
222  ec_master_t *master = slave->master;
223  ec_soe_request_t *request = fsm->request;
224 
225  data = ec_slave_mbox_prepare_send(slave, datagram, EC_MBOX_TYPE_SOE,
226  EC_SOE_SIZE);
227  if (IS_ERR(data)) {
228  return PTR_ERR(data);
229  }
230 
231  EC_WRITE_U8(data, OPCODE_READ_REQUEST | (request->drive_no & 0x07) << 5);
232  EC_WRITE_U8(data + 1, 1 << 6); // request value
233  EC_WRITE_U16(data + 2, request->idn);
234 
235  if (master->debug_level) {
236  EC_SLAVE_DBG(slave, 0, "SCC read request:\n");
237  ec_print_data(data, EC_SOE_SIZE);
238  }
239 
240  fsm->request->jiffies_sent = jiffies;
241  fsm->state = ec_fsm_soe_read_request;
242 
243  return 0;
244 }

281 void ec_fsm_soe_read_request(
282  ec_fsm_soe_t *fsm,
283  ec_datagram_t *datagram
284  )
285 {
286  ec_slave_t *slave = fsm->slave;
287  unsigned long diff_ms;
288 
289  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
290  if (ec_fsm_soe_prepare_read(fsm, datagram)) {
291  fsm->state = ec_fsm_soe_error;
292  ec_fsm_soe_print_error(fsm);
293  }
294  return;
295  }
296 
297  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
298  fsm->state = ec_fsm_soe_error;
299  EC_SLAVE_ERR(slave, "Failed to receive SoE read request: ");
300  ec_datagram_print_state(fsm->datagram);
301  ec_fsm_soe_print_error(fsm);
302  return;
303  }
304 
305  diff_ms = (jiffies - fsm->request->jiffies_sent) * 1000 / HZ;
306 
307  if (fsm->datagram->working_counter != 1) {
308  if (!fsm->datagram->working_counter) {
309  if (diff_ms < EC_SOE_RESPONSE_TIMEOUT) {
310  // no response; send request datagram again
311  if (ec_fsm_soe_prepare_read(fsm, datagram)) {
312  fsm->state = ec_fsm_soe_error;
313  ec_fsm_soe_print_error(fsm);
314  }
315  return;
316  }
317  }
318  fsm->state = ec_fsm_soe_error;
319  EC_SLAVE_ERR(slave, "Reception of SoE read request"
320  " failed after %lu ms: ", diff_ms);
321  ec_datagram_print_wc_error(fsm->datagram);
322  ec_fsm_soe_print_error(fsm);
323  return;
324  }
325 
326  fsm->jiffies_start = fsm->datagram->jiffies_sent;
327  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
328  fsm->retries = EC_FSM_RETRIES;
329  fsm->state = ec_fsm_soe_read_check;
330 }

336 void ec_fsm_soe_read_check(
337  ec_fsm_soe_t *fsm,
338  ec_datagram_t *datagram
339  )
340 {
341  ec_slave_t *slave = fsm->slave;
342 
343  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
344  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
345  return;
346  }
347 
348  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
349  fsm->state = ec_fsm_soe_error;
350  EC_SLAVE_ERR(slave, "Failed to receive SoE mailbox check datagram: ");
351  ec_datagram_print_state(fsm->datagram);
352  ec_fsm_soe_print_error(fsm);
353  return;
354  }
355 
356  if (fsm->datagram->working_counter != 1) {
357  fsm->state = ec_fsm_soe_error;
358  EC_SLAVE_ERR(slave, "Reception of SoE mailbox check"
359  " datagram failed: ");
360  ec_datagram_print_wc_error(fsm->datagram);
361  ec_fsm_soe_print_error(fsm);
362  return;
363  }
364 
365  if (!ec_slave_mbox_check(fsm->datagram)) {
366  unsigned long diff_ms =
367  (fsm->datagram->jiffies_received - fsm->jiffies_start) *
368  1000 / HZ;
369  if (diff_ms >= EC_SOE_RESPONSE_TIMEOUT) {
370  fsm->state = ec_fsm_soe_error;
371  EC_SLAVE_ERR(slave, "Timeout after %lu ms while waiting for"
372  " read response.\n", diff_ms);
373  ec_fsm_soe_print_error(fsm);
374  return;
375  }
376 
377  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
378  fsm->retries = EC_FSM_RETRIES;
379  return;
380  }
381 
382  // Fetch response
383  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
384  fsm->retries = EC_FSM_RETRIES;
385  fsm->state = ec_fsm_soe_read_response;
386 }

392 void ec_fsm_soe_read_response(
393  ec_fsm_soe_t *fsm,
394  ec_datagram_t *datagram
395  )
396 {
397  ec_slave_t *slave = fsm->slave;
398  ec_master_t *master = slave->master;
399  uint8_t *data, mbox_prot, header, opcode, incomplete, error_flag,
400  value_included;
401  size_t rec_size, data_size;
402  ec_soe_request_t *req = fsm->request;
403 
404  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
405  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
406  return;
407  }
408 
409  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
410  fsm->state = ec_fsm_soe_error;
411  EC_SLAVE_ERR(slave, "Failed to receive SoE read response datagram: ");
412  ec_datagram_print_state(fsm->datagram);
413  ec_fsm_soe_print_error(fsm);
414  return;
415  }
416 
417  if (fsm->datagram->working_counter != 1) {
418  fsm->state = ec_fsm_soe_error;
419  EC_SLAVE_ERR(slave, "Reception of SoE read response failed: ");
420  ec_datagram_print_wc_error(fsm->datagram);
421  ec_fsm_soe_print_error(fsm);
422  return;
423  }
424 
425  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
426  if (IS_ERR(data)) {
427  fsm->state = ec_fsm_soe_error;
428  ec_fsm_soe_print_error(fsm);
429  return;
430  }
431 
432  if (master->debug_level) {
433  EC_SLAVE_DBG(slave, 0, "SCC read response:\n");
434  ec_print_data(data, rec_size);
435  }
436 
437  if (mbox_prot != EC_MBOX_TYPE_SOE) {
438  fsm->state = ec_fsm_soe_error;
439  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
440  mbox_prot);
441  ec_fsm_soe_print_error(fsm);
442  return;
443  }
444 
445  if (rec_size < EC_SOE_SIZE) {
446  fsm->state = ec_fsm_soe_error;
447  EC_SLAVE_ERR(slave, "Received currupted SoE read response"
448  " (%zu bytes)!\n", rec_size);
449  ec_print_data(data, rec_size);
450  ec_fsm_soe_print_error(fsm);
451  return;
452  }
453 
454  header = EC_READ_U8(data);
455  opcode = header & 0x7;
456  incomplete = (header >> 3) & 1;
457  error_flag = (header >> 4) & 1;
458 
459  if (opcode != OPCODE_READ_RESPONSE) {
460  EC_SLAVE_ERR(slave, "Received no read response (opcode %x).\n",
461  opcode);
462  ec_print_data(data, rec_size);
463  ec_fsm_soe_print_error(fsm);
464  fsm->state = ec_fsm_soe_error;
465  return;
466  }
467 
468  if (error_flag) {
469  req->error_code = EC_READ_U16(data + rec_size - 2);
470  EC_SLAVE_ERR(slave, "Received error response:\n");
471  ec_print_soe_error(slave, req->error_code);
472  ec_fsm_soe_print_error(fsm);
473  fsm->state = ec_fsm_soe_error;
474  return;
475  } else {
476  req->error_code = 0x0000;
477  }
478 
479  value_included = (EC_READ_U8(data + 1) >> 6) & 1;
480  if (!value_included) {
481  EC_SLAVE_ERR(slave, "No value included!\n");
482  ec_fsm_soe_print_error(fsm);
483  fsm->state = ec_fsm_soe_error;
484  return;
485  }
486 
487  data_size = rec_size - EC_SOE_SIZE;
488  if (ec_soe_request_append_data(req,
489  data + EC_SOE_SIZE, data_size)) {
490  fsm->state = ec_fsm_soe_error;
491  ec_fsm_soe_print_error(fsm);
492  return;
493  }
494 
495  if (incomplete) {
496  EC_SLAVE_DBG(slave, 1, "SoE data incomplete. Waiting for fragment"
497  " at offset %zu.\n", req->data_size);
498  fsm->jiffies_start = fsm->datagram->jiffies_sent;
499  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
500  fsm->retries = EC_FSM_RETRIES;
501  fsm->state = ec_fsm_soe_read_check;
502  } else {
503  if (master->debug_level) {
504  EC_SLAVE_DBG(slave, 0, "IDN data:\n");
505  ec_print_data(req->data, req->data_size);
506  }
507 
508  fsm->state = ec_fsm_soe_end; // success
509  }
510 }

518 void ec_fsm_soe_write_next_fragment(
519  ec_fsm_soe_t *fsm,
520  ec_datagram_t *datagram
521  )
522 {
523  ec_slave_t *slave = fsm->slave;
524  ec_master_t *master = slave->master;
525  ec_soe_request_t *req = fsm->request;
526  uint8_t incomplete, *data;
527  size_t header_size, max_fragment_size, remaining_size;
528  uint16_t fragments_left;
529 
530  header_size = EC_MBOX_HEADER_SIZE + EC_SOE_SIZE;
531  if (slave->configured_rx_mailbox_size <= header_size) {
532  EC_SLAVE_ERR(slave, "Mailbox size (%u) too small for SoE write.\n",
533  slave->configured_rx_mailbox_size);
534  fsm->state = ec_fsm_soe_error;
535  ec_fsm_soe_print_error(fsm);
536  return;
537  }
538 
539  remaining_size = req->data_size - fsm->offset;
540  max_fragment_size = slave->configured_rx_mailbox_size - header_size;
541  incomplete = remaining_size > max_fragment_size;
542  fsm->fragment_size = incomplete ? max_fragment_size : remaining_size;
543  fragments_left = remaining_size / fsm->fragment_size - 1;
544  if (remaining_size % fsm->fragment_size) {
545  fragments_left++;
546  }
547 
548  data = ec_slave_mbox_prepare_send(slave, datagram, EC_MBOX_TYPE_SOE,
549  EC_SOE_SIZE + fsm->fragment_size);
550  if (IS_ERR(data)) {
551  fsm->state = ec_fsm_soe_error;
552  ec_fsm_soe_print_error(fsm);
553  return;
554  }
555 
556  EC_WRITE_U8(data, OPCODE_WRITE_REQUEST | incomplete << 3 |
557  (req->drive_no & 0x07) << 5);
558  EC_WRITE_U8(data + 1, 1 << 6); // only value included
559  EC_WRITE_U16(data + 2, incomplete ? fragments_left : req->idn);
560  memcpy(data + 4, req->data + fsm->offset, fsm->fragment_size);
561 
562  if (master->debug_level) {
563  EC_SLAVE_DBG(slave, 0, "SCC write request:\n");
564  ec_print_data(data, EC_SOE_SIZE + fsm->fragment_size);
565  }
566 
567  req->jiffies_sent = jiffies;
568  fsm->state = ec_fsm_soe_write_request;
569 }

672  if (fsm->datagram->working_counter != 1) {
673  fsm->state = ec_fsm_soe_error;
674  EC_SLAVE_ERR(slave, "Reception of SoE write request datagram: ");
675  ec_datagram_print_wc_error(fsm->datagram);
676  ec_fsm_soe_print_error(fsm);
677  return;
678  }
679 
680  if (!ec_slave_mbox_check(fsm->datagram)) {
681  unsigned long diff_ms =
682  (datagram->jiffies_received - fsm->jiffies_start) * 1000 / HZ;
683  if (diff_ms >= EC_SOE_RESPONSE_TIMEOUT) {
684  fsm->state = ec_fsm_soe_error;
685  EC_SLAVE_ERR(slave, "Timeout after %lu ms while waiting"
686  " for write response.\n", diff_ms);
687  ec_fsm_soe_print_error(fsm);
688  return;
689  }
690 
691  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
692  fsm->retries = EC_FSM_RETRIES;
693  return;
694  }
695 
696  // Fetch response
697  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
698  fsm->retries = EC_FSM_RETRIES;
699  fsm->state = ec_fsm_soe_write_response;
700 }

702 /*****************************************************************************/
703 
706 void ec_fsm_soe_write_response(
707  ec_fsm_soe_t *fsm,
708  ec_datagram_t *datagram
709  )
710 {
711  ec_slave_t *slave = fsm->slave;
712  ec_master_t *master = slave->master;
713  ec_soe_request_t *req = fsm->request;
714  uint8_t *data, mbox_prot, opcode, error_flag;
715  uint16_t idn;
716  size_t rec_size;
717 
718  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
719  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
720  return; // FIXME: request again?
721  }
722 
723  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
724  fsm->state = ec_fsm_soe_error;
725  EC_SLAVE_ERR(slave, "Failed to receive SoE write"
726  " response datagram: ");
727  ec_datagram_print_state(fsm->datagram);
728  ec_fsm_soe_print_error(fsm);
729  return;
730  }
731 
732  if (fsm->datagram->working_counter != 1) {
733  fsm->state = ec_fsm_soe_error;
734  EC_SLAVE_ERR(slave, "Reception of SoE write response failed: ");
735  ec_datagram_print_wc_error(fsm->datagram);
736  ec_fsm_soe_print_error(fsm);
737  return;
738  }
739 
740  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
741  if (IS_ERR(data)) {
742  fsm->state = ec_fsm_soe_error;
743  ec_fsm_soe_print_error(fsm);
744  return;
745  }
746 
747  if (master->debug_level) {
748  EC_SLAVE_DBG(slave, 0, "SCC write response:\n");
749  ec_print_data(data, rec_size);
750  }
751 
752  if (mbox_prot != EC_MBOX_TYPE_SOE) {
753  fsm->state = ec_fsm_soe_error;
754  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
755  mbox_prot);
756  ec_fsm_soe_print_error(fsm);
757  return;
758  }
759 
760  if (rec_size < EC_SOE_SIZE) {
761  fsm->state = ec_fsm_soe_error;
762  EC_SLAVE_ERR(slave, "Received corrupted SoE write response"
763  " (%zu bytes)!\n", rec_size);
764  ec_print_data(data, rec_size);
765  ec_fsm_soe_print_error(fsm);
766  return;
767  }
768 
769  opcode = EC_READ_U8(data) & 0x7;
770  if (opcode != OPCODE_WRITE_RESPONSE) {
771  EC_SLAVE_ERR(slave, "Received no write response"
772  " (opcode %x).\n", opcode);
773  ec_print_data(data, rec_size);
774  ec_fsm_soe_print_error(fsm);
775  fsm->state = ec_fsm_soe_error;
776  return;
777  }
778 
779  idn = EC_READ_U16(data + 2);
780  if (idn != req->idn) {
781  EC_SLAVE_ERR(slave, "Received response for"
782  " wrong IDN 0x%04x.\n", idn);
783  ec_print_data(data, rec_size);
784  ec_fsm_soe_print_error(fsm);
785  fsm->state = ec_fsm_soe_error;
786  return;
787  }
788 
789  error_flag = (EC_READ_U8(data) >> 4) & 1;
790  if (error_flag) {
791  if (rec_size < EC_SOE_SIZE + 2) {
792  EC_SLAVE_ERR(slave, "Received corrupted error response"
793  " - error flag set, but received size is %zu.\n",
794  rec_size);
795  } else {
796  req->error_code = EC_READ_U16(data + EC_SOE_SIZE);
797  EC_SLAVE_ERR(slave, "Received error response:\n");
798  ec_print_soe_error(slave, req->error_code);
799  }
800  ec_print_data(data, rec_size);
801  ec_fsm_soe_print_error(fsm);
802  fsm->state = ec_fsm_soe_error;
803  return;
804  } else {
805  req->error_code = 0x0000;
806  }
807 
808  fsm->offset += fsm->fragment_size;
809 
810  if (fsm->offset < req->data_size) {
811  fsm->retries = EC_FSM_RETRIES;
812  ec_fsm_soe_write_next_fragment(fsm, datagram);
813  } else {
814  fsm->state = ec_fsm_soe_end; // success
815  }
816 }
