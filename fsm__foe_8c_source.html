126 int ec_fsm_foe_exec(
127  ec_fsm_foe_t *fsm,
128  ec_datagram_t *datagram
129  )
130 {
131  int datagram_used = 0;
132 
133  if (fsm->datagram &&
134  (fsm->datagram->state == EC_DATAGRAM_INIT ||
135  fsm->datagram->state == EC_DATAGRAM_QUEUED ||
136  fsm->datagram->state == EC_DATAGRAM_SENT)) {
137  // datagram not received yet
138  return datagram_used;
139  }
140 
141  fsm->state(fsm, datagram);
142 
143  datagram_used =
144  fsm->state != ec_fsm_foe_end && fsm->state != ec_fsm_foe_error;
145 
146  if (datagram_used) {
147  fsm->datagram = datagram;
148  } else {
149  fsm->datagram = NULL;
150  }
151 
152  return datagram_used;
153 }

233 int ec_foe_prepare_data_send(
234  ec_fsm_foe_t *fsm,
235  ec_datagram_t *datagram
236  )
237 {
238  size_t remaining_size, current_size;
239  uint8_t *data;
240 
241  remaining_size = fsm->tx_buffer_size - fsm->tx_buffer_offset;
242 
243  if (remaining_size < fsm->slave->configured_tx_mailbox_size
244  - EC_MBOX_HEADER_SIZE - EC_FOE_HEADER_SIZE) {
245  current_size = remaining_size;
246  fsm->tx_last_packet = 1;
247  } else {
248  current_size = fsm->slave->configured_tx_mailbox_size
249  - EC_MBOX_HEADER_SIZE - EC_FOE_HEADER_SIZE;
250  }
251 
252  data = ec_slave_mbox_prepare_send(fsm->slave,
253  datagram, EC_MBOX_TYPE_FOE, current_size + EC_FOE_HEADER_SIZE);
254  if (IS_ERR(data)) {
255  return -1;
256  }
257 
258  EC_WRITE_U16(data, EC_FOE_OPCODE_DATA); // OpCode = DataBlock req.
259  EC_WRITE_U32(data + 2, fsm->tx_packet_no); // PacketNo, Password
260 #ifdef DEBUG_FOE
261  EC_SLAVE_DBG(fsm->slave, 0, "sending opcode %u packet %u\n",
262  EC_FOE_OPCODE_DATA, fsm->tx_packet_no);
263 #endif
264 
265  memcpy(data + EC_FOE_HEADER_SIZE,
266  fsm->tx_buffer + fsm->tx_buffer_offset, current_size);
267  fsm->tx_current_size = current_size;
268 
269  return 0;
270 }

349 void ec_fsm_foe_state_ack_check(
350  ec_fsm_foe_t *fsm,
351  ec_datagram_t *datagram
352  )
353 {
354  ec_slave_t *slave = fsm->slave;
355 
356 #ifdef DEBUG_FOE
357  EC_SLAVE_DBG(fsm->slave, 0, "%s()\n", __func__);
358 #endif
359 
360  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
361  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
362  EC_SLAVE_ERR(slave, "Failed to receive FoE mailbox check datagram: ");
363  ec_datagram_print_state(fsm->datagram);
364  return;
365  }
366 
367  if (fsm->datagram->working_counter != 1) {
368  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
369  EC_SLAVE_ERR(slave, "Reception of FoE mailbox check datagram"
370  " failed: ");
371  ec_datagram_print_wc_error(fsm->datagram);
372  return;
373  }
374 
375  if (!ec_slave_mbox_check(fsm->datagram)) {
376  // slave did not put anything in the mailbox yet
377  if (time_after(fsm->datagram->jiffies_received,
378  fsm->jiffies_start + EC_FSM_FOE_TIMEOUT_JIFFIES)) {
379  ec_foe_set_tx_error(fsm, FOE_TIMEOUT_ERROR);
380  EC_SLAVE_ERR(slave, "Timeout while waiting for ack response.\n");
381  return;
382  }
383 
384  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
385  fsm->retries = EC_FSM_RETRIES;
386  return;
387  }
388 
389  // Fetch response
390  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
391 
392  fsm->retries = EC_FSM_RETRIES;
393  fsm->state = ec_fsm_foe_state_ack_read;
394 }

400 void ec_fsm_foe_state_ack_read(
401  ec_fsm_foe_t *fsm,
402  ec_datagram_t *datagram
403  )
404 {
405  ec_slave_t *slave = fsm->slave;
406  uint8_t *data, mbox_prot;
407  uint8_t opCode;
408  size_t rec_size;
409 
410 #ifdef DEBUG_FOE
411  EC_SLAVE_DBG(fsm->slave, 0, "%s()\n", __func__);
412 #endif
413 
414  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
415  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
416  EC_SLAVE_ERR(slave, "Failed to receive FoE ack response datagram: ");
417  ec_datagram_print_state(fsm->datagram);
418  return;
419  }
420 
421  if (fsm->datagram->working_counter != 1) {
422  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
423  EC_SLAVE_ERR(slave, "Reception of FoE ack response failed: ");
424  ec_datagram_print_wc_error(fsm->datagram);
425  return;
426  }
427 
428  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
429  if (IS_ERR(data)) {
430  ec_foe_set_tx_error(fsm, FOE_PROT_ERROR);
431  return;
432  }
433 
434  if (mbox_prot != EC_MBOX_TYPE_FOE) {
435  ec_foe_set_tx_error(fsm, FOE_MBOX_PROT_ERROR);
436  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
437  mbox_prot);
438  return;
439  }
440 
441  opCode = EC_READ_U8(data);
442 #ifdef DEBUG_FOE
443  EC_SLAVE_DBG(fsm->slave, 0, "received opcode %u\n", opCode);
444 #endif
445 
446  if (opCode == EC_FOE_OPCODE_BUSY) {
447  // slave not ready
448  if (ec_foe_prepare_data_send(fsm, datagram)) {
449  ec_foe_set_tx_error(fsm, FOE_PROT_ERROR);
450  EC_SLAVE_ERR(slave, "Slave is busy.\n");
451  return;
452  }
453  fsm->state = ec_fsm_foe_state_data_sent;
454  return;
455  }
456 
457  if (opCode == EC_FOE_OPCODE_ERR) {
458  fsm->request->error_code = EC_READ_U32(data + 2);
459  EC_SLAVE_ERR(slave, "Received FoE Error Request (code 0x%08x).\n",
460  fsm->request->error_code);
461  if (rec_size > 6 && data[6]) {
462  uint8_t text[256];
463  strncpy(text, data + 6, min(rec_size - 6, sizeof(text)));
464  text[sizeof(text)-1] = 0;
465  EC_SLAVE_ERR(slave, "FoE Error Text: %s\n", text);
466  }
467  ec_foe_set_tx_error(fsm, FOE_OPCODE_ERROR);
468  return;
469  }
470 
471  if (opCode == EC_FOE_OPCODE_ACK) {
472  fsm->tx_packet_no++;
473  fsm->tx_buffer_offset += fsm->tx_current_size;
474 
475  if (fsm->tx_last_packet) {
476  fsm->state = ec_fsm_foe_end;
477  return;
478  }
479 
480  if (ec_foe_prepare_data_send(fsm, datagram)) {
481  ec_foe_set_tx_error(fsm, FOE_PROT_ERROR);
482  return;
483  }
484  fsm->state = ec_fsm_foe_state_data_sent;
485  return;
486  }
487  ec_foe_set_tx_error(fsm, FOE_ACK_ERROR);
488 }

744  }
745 
746  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
747  fsm->retries = EC_FSM_RETRIES;
748  return;
749  }

751  // Fetch response
752  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.

754  fsm->retries = EC_FSM_RETRIES;

762 void ec_fsm_foe_state_data_read(
763  ec_fsm_foe_t *fsm,
764  ec_datagram_t *datagram
765  )
766 {
767  size_t rec_size;
768  uint32_t packet_no;
769  uint8_t *data, opCode, mbox_prot;
770 
771  ec_slave_t *slave = fsm->slave;
772 
773 #ifdef DEBUG_FOE
774  EC_SLAVE_DBG(fsm->slave, 0, "%s()\n", __func__);
775 #endif
776 
777  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
778  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
779  EC_SLAVE_ERR(slave, "Failed to receive FoE DATA READ datagram: ");
780  ec_datagram_print_state(fsm->datagram);
781  return;
782  }
783 
784  if (fsm->datagram->working_counter != 1) {
785  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
786  EC_SLAVE_ERR(slave, "Reception of FoE DATA READ failed: ");
787  ec_datagram_print_wc_error(fsm->datagram);
788  return;
789  }
790 
791  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
792  if (IS_ERR(data)) {
793  ec_foe_set_rx_error(fsm, FOE_MBOX_FETCH_ERROR);
794  return;
795  }
796 
797  if (mbox_prot != EC_MBOX_TYPE_FOE) {
798  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
799  mbox_prot);
800  ec_foe_set_rx_error(fsm, FOE_PROT_ERROR);
801  return;
802  }
803 
804  opCode = EC_READ_U8(data);
805 #ifdef DEBUG_FOE
806  EC_SLAVE_DBG(fsm->slave, 0, "received opcode %u\n", opCode);
807 #endif
808 
809  if (opCode == EC_FOE_OPCODE_BUSY) {
810  fsm->rx_expected_packet_no--;
811  if (ec_foe_prepare_send_ack(fsm, datagram)) {
812  ec_foe_set_rx_error(fsm, FOE_PROT_ERROR);
813  } else {
814  fsm->state = ec_fsm_foe_state_sent_ack;
815  }
816 #ifdef DEBUG_FOE
817  EC_SLAVE_DBG(fsm->slave, 0, "%s() busy. Next pkt %u\n", __func__,
818  fsm->rx_expected_packet_no);
819 #endif
820  return;
821  }
822 
823  if (opCode == EC_FOE_OPCODE_ERR) {
824  fsm->request->error_code = EC_READ_U32(data + 2);
825  EC_SLAVE_ERR(slave, "Received FoE Error Request (code 0x%08x).\n",
826  fsm->request->error_code);
827  if (rec_size > 6 && data[6]) {
828  uint8_t text[256];
829  strncpy(text, data + 6, min(rec_size - 6, sizeof(text)));
830  text[sizeof(text)-1] = 0;
831  EC_SLAVE_ERR(slave, "FoE Error Text: %s\n", text);
832  }
833  ec_foe_set_rx_error(fsm, FOE_OPCODE_ERROR);
834  return;
835  }
836 
837  if (opCode != EC_FOE_OPCODE_DATA) {
838  EC_SLAVE_ERR(slave, "Received OPCODE %x, expected %x.\n",
839  opCode, EC_FOE_OPCODE_DATA);
840  fsm->request->error_code = 0x00000000;
841  ec_foe_set_rx_error(fsm, FOE_OPCODE_ERROR);
842  return;
843  }
844 
845  packet_no = EC_READ_U32(data + 2);
846  if (packet_no != fsm->rx_expected_packet_no) {
847  EC_SLAVE_ERR(slave, "Received packet number %u, expected %u.\n",
848  packet_no, fsm->rx_expected_packet_no);
849  ec_foe_set_rx_error(fsm, FOE_PACKETNO_ERROR);
850  return;
851  }
852 
853  rec_size -= EC_FOE_HEADER_SIZE;
854 
855  if (fsm->rx_buffer_size >= fsm->rx_buffer_offset + rec_size) {
856  memcpy(fsm->rx_buffer + fsm->rx_buffer_offset,
857  data + EC_FOE_HEADER_SIZE, rec_size);
858  fsm->rx_buffer_offset += rec_size;
859  }
860 
861  fsm->rx_last_packet =
862  (rec_size + EC_MBOX_HEADER_SIZE + EC_FOE_HEADER_SIZE
863  != slave->configured_rx_mailbox_size);
864 
865  if (fsm->rx_last_packet ||
866  (slave->configured_rx_mailbox_size - EC_MBOX_HEADER_SIZE
867  - EC_FOE_HEADER_SIZE + fsm->rx_buffer_offset)
868  <= fsm->rx_buffer_size) {
869  // either it was the last packet or a new packet will fit into the
870  // delivered buffer
871 #ifdef DEBUG_FOE
872  EC_SLAVE_DBG(fsm->slave, 0, "last_packet=true\n");
873 #endif
874  if (ec_foe_prepare_send_ack(fsm, datagram)) {
875  ec_foe_set_rx_error(fsm, FOE_RX_DATA_ACK_ERROR);
876  return;
877  }
878 
879  fsm->state = ec_fsm_foe_state_sent_ack;
880  }
881  else {
882  // no more data fits into the delivered buffer
883  // ... wait for new read request
884  EC_SLAVE_ERR(slave, "Data do not fit in receive buffer!\n");
885  printk(" rx_buffer_size = %d\n", fsm->rx_buffer_size);
886  printk("rx_buffer_offset = %d\n", fsm->rx_buffer_offset);
887  printk(" rec_size = %zd\n", rec_size);
888  printk(" rx_mailbox_size = %d\n", slave->configured_rx_mailbox_size);
889  printk(" rx_last_packet = %d\n", fsm->rx_last_packet);
890  fsm->request->result = FOE_READY;
891  }
892 }
