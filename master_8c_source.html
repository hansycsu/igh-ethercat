682 int ec_master_enter_operation_phase(
683  ec_master_t *master
684  )
685 {
686  int ret = 0;
687  ec_slave_t *slave;
688 
689  EC_MASTER_DBG(master, 1, "IDLE -> OPERATION.\n");
690 
691  down(&master->config_sem);
692  if (master->config_busy) {
693  up(&master->config_sem);
694 
695  // wait for slave configuration to complete
696  ret = wait_event_interruptible(master->config_queue,
697  !master->config_busy);
698  if (ret) {
699  EC_MASTER_INFO(master, "Finishing slave configuration"
700  " interrupted by signal.\n");
701  goto out_return;
702  }
703 
704  EC_MASTER_DBG(master, 1, "Waiting for pending slave"
705  " configuration returned.\n");
706  } else {
707  up(&master->config_sem);
708  }
709 
710  down(&master->scan_sem);
711  master->allow_scan = 0; // 'lock' the slave list
712  if (!master->scan_busy) {
713  up(&master->scan_sem);
714  } else {
715  up(&master->scan_sem);
716 
717  // wait for slave scan to complete
718  ret = wait_event_interruptible(master->scan_queue,
719  !master->scan_busy);
720  if (ret) {
721  EC_MASTER_INFO(master, "Waiting for slave scan"
722  " interrupted by signal.\n");
723  goto out_allow;
724  }
725 
726  EC_MASTER_DBG(master, 1, "Waiting for pending"
727  " slave scan returned.\n");
728  }
729 
730  // set states for all slaves
731  for (slave = master->slaves;
732  slave < master->slaves + master->slave_count;
733  slave++) {
734  ec_slave_request_state(slave, EC_SLAVE_STATE_PREOP);
735  }
736 
737  master->phase = EC_OPERATION;
738  master->app_send_cb = NULL;
739  master->app_receive_cb = NULL;
740  master->app_cb_data = NULL;
741  return ret;
742 
743 out_allow:
744  master->allow_scan = 1;
745 out_return:
746  return ret;
747 }

777  )
778 {
779  ec_datagram_t *datagram;
780  size_t queue_size = 0, new_queue_size = 0;
781 #if DEBUG_INJECT
782  unsigned int datagram_count = 0;
783 #endif
784 
785  if (master->ext_ring_idx_rt == master->ext_ring_idx_fsm) {
786  // nothing to inject
787  return;
788  }
789 
790  list_for_each_entry(datagram, &master->datagram_queue, queue) {
791  if (datagram->state == EC_DATAGRAM_QUEUED) {
792  queue_size += datagram->data_size;
793  }
794  }
795 
796 #if DEBUG_INJECT
797  EC_MASTER_DBG(master, 1, "Injecting datagrams, queue_size=%zu\n",
798  queue_size);
799 #endif
800 
801  while (master->ext_ring_idx_rt != master->ext_ring_idx_fsm) {
802  datagram = &master->ext_datagram_ring[master->ext_ring_idx_rt];
803 
804  if (datagram->state != EC_DATAGRAM_INIT) {
805  // skip datagram
806  master->ext_ring_idx_rt =
807  (master->ext_ring_idx_rt + 1) % EC_EXT_RING_SIZE;
808  continue;
809  }
810 
811  new_queue_size = queue_size + datagram->data_size;
812  if (new_queue_size <= master->max_queue_size) {
813 #if DEBUG_INJECT
814  EC_MASTER_DBG(master, 1, "Injecting datagram %s"
815  " size=%zu, queue_size=%zu\n", datagram->name,
816  datagram->data_size, new_queue_size);
817  datagram_count++;
818 #endif
819 #ifdef EC_HAVE_CYCLES
820  datagram->cycles_sent = 0;
821 #endif
822  datagram->jiffies_sent = 0;
823  ec_master_queue_datagram(master, datagram);
824  queue_size = new_queue_size;
825  }
826  else if (datagram->data_size > master->max_queue_size) {
827  datagram->state = EC_DATAGRAM_ERROR;
828  EC_MASTER_ERR(master, "External datagram %s is too large,"
829  " size=%zu, max_queue_size=%zu\n",
830  datagram->name, datagram->data_size,
831  master->max_queue_size);
832  }
833  else { // datagram does not fit in the current cycle
834 #ifdef EC_HAVE_CYCLES
835  cycles_t cycles_now = get_cycles();
836 
837  if (cycles_now - datagram->cycles_sent
838  > ext_injection_timeout_cycles)
839 #else
840  if (jiffies - datagram->jiffies_sent
841  > ext_injection_timeout_jiffies)
842 #endif
843  {
844 #if defined EC_RT_SYSLOG || DEBUG_INJECT
845  unsigned int time_us;
846 #endif
847 
848  datagram->state = EC_DATAGRAM_ERROR;
849 
850 #if defined EC_RT_SYSLOG || DEBUG_INJECT
851 #ifdef EC_HAVE_CYCLES
852  time_us = (unsigned int)
853  ((cycles_now - datagram->cycles_sent) * 1000LL)
854  / cpu_khz;
855 #else
856  time_us = (unsigned int)
857  ((jiffies - datagram->jiffies_sent) * 1000000 / HZ);
858 #endif
859  EC_MASTER_ERR(master, "Timeout %u us: Injecting"
860  " external datagram %s size=%zu,"
861  " max_queue_size=%zu\n", time_us, datagram->name,
862  datagram->data_size, master->max_queue_size);
863 #endif
864  }
865  else {
866 #if DEBUG_INJECT
867  EC_MASTER_DBG(master, 1, "Deferred injecting"
868  " external datagram %s size=%u, queue_size=%u\n",
869  datagram->name, datagram->data_size, queue_size);
870 #endif
871  break;
872  }
873  }
874 
875  master->ext_ring_idx_rt =
876  (master->ext_ring_idx_rt + 1) % EC_EXT_RING_SIZE;
877  }
878 
879 #if DEBUG_INJECT
880  EC_MASTER_DBG(master, 1, "Injected %u datagrams.\n", datagram_count);
881 #endif
882 }

927  ec_datagram_t *datagram
928  )
929 {

944 #endif
945  datagram->state = EC_DATAGRAM_QUEUED;
946  return;
947  }
948  }
949 
950  list_add_tail(&datagram->queue, &master->datagram_queue);
951  datagram->state = EC_DATAGRAM_QUEUED;

973 size_t ec_master_send_datagrams(
974  ec_master_t *master,
975  ec_device_index_t device_index
976  )
977 {
978  ec_datagram_t *datagram, *next;
979  size_t datagram_size;
980  uint8_t *frame_data, *cur_data = NULL;
981  void *follows_word;
982 #ifdef EC_HAVE_CYCLES
983  cycles_t cycles_start, cycles_sent, cycles_end;
984 #endif
985  unsigned long jiffies_sent;
986  unsigned int frame_count, more_datagrams_waiting;
987  struct list_head sent_datagrams;
988  size_t sent_bytes = 0;
989 
990 #ifdef EC_HAVE_CYCLES
991  cycles_start = get_cycles();
992 #endif
993  frame_count = 0;
994  INIT_LIST_HEAD(&sent_datagrams);
995 
996  EC_MASTER_DBG(master, 2, "%s(device_index = %u)\n",
997  __func__, device_index);
998 
999  do {
1000  frame_data = NULL;
1001  follows_word = NULL;
1002  more_datagrams_waiting = 0;
1003 
1004  // fill current frame with datagrams
1005  list_for_each_entry(datagram, &master->datagram_queue, queue) {
1006  if (datagram->state != EC_DATAGRAM_QUEUED ||
1007  datagram->device_index != device_index) {
1008  continue;
1009  }
1010 
1011  if (!frame_data) {
1012  // fetch pointer to transmit socket buffer
1013  frame_data =
1014  ec_device_tx_data(&master->devices[device_index]);
1015  cur_data = frame_data + EC_FRAME_HEADER_SIZE;
1016  }
1017 
1018  // does the current datagram fit in the frame?
1019  datagram_size = EC_DATAGRAM_HEADER_SIZE + datagram->data_size
1020  + EC_DATAGRAM_FOOTER_SIZE;
1021  if (cur_data - frame_data + datagram_size > ETH_DATA_LEN) {
1022  more_datagrams_waiting = 1;
1023  break;
1024  }
1025 
1026  list_add_tail(&datagram->sent, &sent_datagrams);
1027  datagram->index = master->datagram_index++;
1028 
1029  EC_MASTER_DBG(master, 2, "Adding datagram 0x%02X\n",
1030  datagram->index);
1031 
1032  // set "datagram following" flag in previous datagram
1033  if (follows_word) {
1034  EC_WRITE_U16(follows_word,
1035  EC_READ_U16(follows_word) | 0x8000);
1036  }
1037 
1038  // EtherCAT datagram header
1039  EC_WRITE_U8 (cur_data, datagram->type);
1040  EC_WRITE_U8 (cur_data + 1, datagram->index);
1041  memcpy(cur_data + 2, datagram->address, EC_ADDR_LEN);
1042  EC_WRITE_U16(cur_data + 6, datagram->data_size & 0x7FF);
1043  EC_WRITE_U16(cur_data + 8, 0x0000);
1044  follows_word = cur_data + 6;
1045  cur_data += EC_DATAGRAM_HEADER_SIZE;
1046 
1047  // EtherCAT datagram data
1048  memcpy(cur_data, datagram->data, datagram->data_size);
1049  cur_data += datagram->data_size;
1050 
1051  // EtherCAT datagram footer
1052  EC_WRITE_U16(cur_data, 0x0000); // reset working counter
1053  cur_data += EC_DATAGRAM_FOOTER_SIZE;
1054  }
1055 
1056  if (list_empty(&sent_datagrams)) {
1057  EC_MASTER_DBG(master, 2, "nothing to send.\n");
1058  break;
1059  }
1060 
1061  // EtherCAT frame header
1062  EC_WRITE_U16(frame_data, ((cur_data - frame_data
1063  - EC_FRAME_HEADER_SIZE) & 0x7FF) | 0x1000);
1064 
1065  // pad frame
1066  while (cur_data - frame_data < ETH_ZLEN - ETH_HLEN)
1067  EC_WRITE_U8(cur_data++, 0x00);
1068 
1069  EC_MASTER_DBG(master, 2, "frame size: %zu\n", cur_data - frame_data);
1070 
1071  // send frame
1072  ec_device_send(&master->devices[device_index],
1073  cur_data - frame_data);
1074  /* preamble and inter-frame gap */
1075  sent_bytes += ETH_HLEN + cur_data - frame_data + ETH_FCS_LEN + 20;
1076 #ifdef EC_HAVE_CYCLES
1077  cycles_sent = get_cycles();
1078 #endif
1079  jiffies_sent = jiffies;
1080 
1081  // set datagram states and sending timestamps
1082  list_for_each_entry_safe(datagram, next, &sent_datagrams, sent) {
1083  datagram->state = EC_DATAGRAM_SENT;
1084 #ifdef EC_HAVE_CYCLES
1085  datagram->cycles_sent = cycles_sent;
1086 #endif
1087  datagram->jiffies_sent = jiffies_sent;
1088  list_del_init(&datagram->sent); // empty list of sent datagrams
1089  }
1090 
1091  frame_count++;
1092  }
1093  while (more_datagrams_waiting && frame_count < EC_TX_RING_SIZE);
1094 
1095 #ifdef EC_HAVE_CYCLES
1096  if (unlikely(master->debug_level > 1)) {
1097  cycles_end = get_cycles();
1098  EC_MASTER_DBG(master, 0, "%s()"
1099  " sent %u frames in %uus.\n", __func__, frame_count,
1100  (unsigned int) (cycles_end - cycles_start) * 1000 / cpu_khz);
1101  }
1102 #endif
1103  return sent_bytes;
1104 }

1114 void ec_master_receive_datagrams(
1115  ec_master_t *master,
1116  ec_device_t *device,
1117  const uint8_t *frame_data,
1118  size_t size
1119  )
1120 {
1121  size_t frame_size, data_size;
1122  uint8_t datagram_type, datagram_index;
1123  unsigned int cmd_follows, matched;
1124  const uint8_t *cur_data;
1125  ec_datagram_t *datagram;
1126 
1127  if (unlikely(size < EC_FRAME_HEADER_SIZE)) {
1128  if (master->debug_level || FORCE_OUTPUT_CORRUPTED) {
1129  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1130  " on %s (size %zu < %u byte):\n",
1131  device->dev->name, size, EC_FRAME_HEADER_SIZE);
1132  ec_print_data(frame_data, size);
1133  }
1134  master->stats.corrupted++;
1135 #ifdef EC_RT_SYSLOG
1136  ec_master_output_stats(master);
1137 #endif
1138  return;
1139  }
1140 
1141  cur_data = frame_data;
1142 
1143  // check length of entire frame
1144  frame_size = EC_READ_U16(cur_data) & 0x07FF;
1145  cur_data += EC_FRAME_HEADER_SIZE;
1146 
1147  if (unlikely(frame_size > size)) {
1148  if (master->debug_level || FORCE_OUTPUT_CORRUPTED) {
1149  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1150  " on %s (invalid frame size %zu for "
1151  "received size %zu):\n", device->dev->name,
1152  frame_size, size);
1153  ec_print_data(frame_data, size);
1154  }
1155  master->stats.corrupted++;
1156 #ifdef EC_RT_SYSLOG
1157  ec_master_output_stats(master);
1158 #endif
1159  return;
1160  }
1161 
1162  cmd_follows = 1;
1163  while (cmd_follows) {
1164  // process datagram header
1165  datagram_type = EC_READ_U8 (cur_data);
1166  datagram_index = EC_READ_U8 (cur_data + 1);
1167  data_size = EC_READ_U16(cur_data + 6) & 0x07FF;
1168  cmd_follows = EC_READ_U16(cur_data + 6) & 0x8000;
1169  cur_data += EC_DATAGRAM_HEADER_SIZE;
1170 
1171  if (unlikely(cur_data - frame_data
1172  + data_size + EC_DATAGRAM_FOOTER_SIZE > size)) {
1173  if (master->debug_level || FORCE_OUTPUT_CORRUPTED) {
1174  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1175  " on %s (invalid data size %zu):\n",
1176  device->dev->name, data_size);
1177  ec_print_data(frame_data, size);
1178  }
1179  master->stats.corrupted++;
1180 #ifdef EC_RT_SYSLOG
1181  ec_master_output_stats(master);
1182 #endif
1183  return;
1184  }
1185 
1186  // search for matching datagram in the queue
1187  matched = 0;
1188  list_for_each_entry(datagram, &master->datagram_queue, queue) {
1189  if (datagram->index == datagram_index
1190  && datagram->state == EC_DATAGRAM_SENT
1191  && datagram->type == datagram_type
1192  && datagram->data_size == data_size) {
1193  matched = 1;
1194  break;
1195  }
1196  }
1197 
1198  // no matching datagram was found
1199  if (!matched) {
1200  master->stats.unmatched++;
1201 #ifdef EC_RT_SYSLOG
1202  ec_master_output_stats(master);
1203 #endif
1204 
1205  if (unlikely(master->debug_level > 0)) {
1206  EC_MASTER_DBG(master, 0, "UNMATCHED datagram:\n");
1207  ec_print_data(cur_data - EC_DATAGRAM_HEADER_SIZE,
1208  EC_DATAGRAM_HEADER_SIZE + data_size
1209  + EC_DATAGRAM_FOOTER_SIZE);
1210 #ifdef EC_DEBUG_RING
1211  ec_device_debug_ring_print(&master->devices[EC_DEVICE_MAIN]);
1212 #endif
1213  }
1214 
1215  cur_data += data_size + EC_DATAGRAM_FOOTER_SIZE;
1216  continue;
1217  }
1218 
1219  if (datagram->type != EC_DATAGRAM_APWR &&
1220  datagram->type != EC_DATAGRAM_FPWR &&
1221  datagram->type != EC_DATAGRAM_BWR &&
1222  datagram->type != EC_DATAGRAM_LWR) {
1223  // copy received data into the datagram memory,
1224  // if something has been read
1225  memcpy(datagram->data, cur_data, data_size);
1226  }
1227  cur_data += data_size;
1228 
1229  // set the datagram's working counter
1230  datagram->working_counter = EC_READ_U16(cur_data);
1231  cur_data += EC_DATAGRAM_FOOTER_SIZE;
1232 
1233  // dequeue the received datagram
1234  datagram->state = EC_DATAGRAM_RECEIVED;
1235 #ifdef EC_HAVE_CYCLES
1236  datagram->cycles_received =
1237  master->devices[EC_DEVICE_MAIN].cycles_poll;
1238 #endif
1239  datagram->jiffies_received =
1240  master->devices[EC_DEVICE_MAIN].jiffies_poll;
1241  list_del_init(&datagram->queue);
1242  }
1243 }

1310 void ec_master_update_device_stats(
1311  ec_master_t *master
1312  )
1313 {
1314  ec_device_stats_t *s = &master->device_stats;
1315  s32 tx_frame_rate, rx_frame_rate, tx_byte_rate, rx_byte_rate, loss_rate;
1316  u64 loss;
1317  unsigned int i, dev_idx;
1318 
1319  // frame statistics
1320  if (likely(jiffies - s->jiffies < HZ)) {
1321  return;
1322  }
1323 
1324  tx_frame_rate = (s->tx_count - s->last_tx_count) * 1000;
1325  rx_frame_rate = (s->rx_count - s->last_rx_count) * 1000;
1326  tx_byte_rate = s->tx_bytes - s->last_tx_bytes;
1327  rx_byte_rate = s->rx_bytes - s->last_rx_bytes;
1328  loss = s->tx_count - s->rx_count;
1329  loss_rate = (loss - s->last_loss) * 1000;
1330 
1331  /* Low-pass filter:
1332  * Y_n = y_(n - 1) + T / tau * (x - y_(n - 1)) | T = 1
1333  * -> Y_n += (x - y_(n - 1)) / tau
1334  */
1335  for (i = 0; i < EC_RATE_COUNT; i++) {
1336  s32 n = rate_intervals[i];
1337  s->tx_frame_rates[i] += (tx_frame_rate - s->tx_frame_rates[i]) / n;
1338  s->rx_frame_rates[i] += (rx_frame_rate - s->rx_frame_rates[i]) / n;
1339  s->tx_byte_rates[i] += (tx_byte_rate - s->tx_byte_rates[i]) / n;
1340  s->rx_byte_rates[i] += (rx_byte_rate - s->rx_byte_rates[i]) / n;
1341  s->loss_rates[i] += (loss_rate - s->loss_rates[i]) / n;
1342  }
1343 
1344  s->last_tx_count = s->tx_count;
1345  s->last_rx_count = s->rx_count;
1346  s->last_tx_bytes = s->tx_bytes;
1347  s->last_rx_bytes = s->rx_bytes;
1348  s->last_loss = loss;
1349 
1350  for (dev_idx = EC_DEVICE_MAIN; dev_idx < ec_master_num_devices(master);
1351  dev_idx++) {
1352  ec_device_update_stats(&master->devices[dev_idx]);
1353  }
1354 
1355  s->jiffies = jiffies;
1356 }
