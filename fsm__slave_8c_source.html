199 void ec_fsm_slave_state_ready(
200  ec_fsm_slave_t *fsm,
201  ec_datagram_t *datagram
202  )
203 {
204  ec_slave_t *slave = fsm->slave;
205  ec_sdo_request_t *req;
206 
207  if (slave->config) {
208  list_for_each_entry(req, &slave->config->sdo_requests, list) {
209  if (req->state == EC_INT_REQUEST_QUEUED || req->state == EC_INT_REQUEST_BUSY) {
210  EC_SLAVE_DBG(slave, 1, "Busy - processing internal SDO request!\n");
211  return;
212  }
213  }
214  }
215 
216  // Check for pending external SDO requests
217  if (ec_fsm_slave_action_process_sdo(fsm, datagram)) {
218  return;
219  }
220 
221  // Check for pending external register requests
222  if (ec_fsm_slave_action_process_reg(fsm, datagram)) {
223  return;
224  }
225 
226  // Check for pending FoE requests
227  if (ec_fsm_slave_action_process_foe(fsm, datagram)) {
228  return;
229  }
230 
231  // Check for pending SoE requests
232  if (ec_fsm_slave_action_process_soe(fsm, datagram)) {
233  return;
234  }
235 
236  // Check for pending EoE IP parameter requests
237  if (ec_fsm_slave_action_process_eoe(fsm, datagram)) {
238  return;
239  }
240 }

248 int ec_fsm_slave_action_process_sdo(
249  ec_fsm_slave_t *fsm,
250  ec_datagram_t *datagram
251  )
252 {
253  ec_slave_t *slave = fsm->slave;
254  ec_sdo_request_t *request;
255 
256  if (list_empty(&slave->sdo_requests)) {
257  return 0;
258  }
259 
260  // take the first request to be processed
261  request = list_entry(slave->sdo_requests.next, ec_sdo_request_t, list);
262  list_del_init(&request->list); // dequeue
263 
264  if (slave->current_state & EC_SLAVE_STATE_ACK_ERR) {
265  EC_SLAVE_WARN(slave, "Aborting SDO request,"
266  " slave has error flag set.\n");
267  request->state = EC_INT_REQUEST_FAILURE;
268  wake_up_all(&slave->master->request_queue);
269  fsm->state = ec_fsm_slave_state_idle;
270  return 0;
271  }
272 
273  if (slave->current_state == EC_SLAVE_STATE_INIT) {
274  EC_SLAVE_WARN(slave, "Aborting SDO request, slave is in INIT.\n");
275  request->state = EC_INT_REQUEST_FAILURE;
276  wake_up_all(&slave->master->request_queue);
277  fsm->state = ec_fsm_slave_state_idle;
278  return 0;
279  }
280 
281  fsm->sdo_request = request;
282  request->state = EC_INT_REQUEST_BUSY;
283 
284  // Found pending SDO request. Execute it!
285  EC_SLAVE_DBG(slave, 1, "Processing SDO request...\n");
286 
287  // Start SDO transfer
288  fsm->state = ec_fsm_slave_state_sdo_request;
289  ec_fsm_coe_transfer(&fsm->fsm_coe, slave, request);
290  ec_fsm_coe_exec(&fsm->fsm_coe, datagram); // execute immediately
291  return 1;
292 }

334 int ec_fsm_slave_action_process_reg(
335  ec_fsm_slave_t *fsm,
336  ec_datagram_t *datagram
337  )
338 {
339  ec_slave_t *slave = fsm->slave;
340  ec_reg_request_t *reg;
341 
342  fsm->reg_request = NULL;
343 
344  if (slave->config) {
345  // search the first internal register request to be processed
346  list_for_each_entry(reg, &slave->config->reg_requests, list) {
347  if (reg->state == EC_INT_REQUEST_QUEUED) {
348  fsm->reg_request = reg;
349  break;
350  }
351  }
352  }
353 
354  if (!fsm->reg_request && !list_empty(&slave->reg_requests)) {
355  // take the first external request to be processed
356  fsm->reg_request =
357  list_entry(slave->reg_requests.next, ec_reg_request_t, list);
358  list_del_init(&fsm->reg_request->list); // dequeue
359  }
360 
361  if (!fsm->reg_request) { // no register request to process
362  return 0;
363  }
364 
365  if (slave->current_state & EC_SLAVE_STATE_ACK_ERR) {
366  EC_SLAVE_WARN(slave, "Aborting register request,"
367  " slave has error flag set.\n");
368  fsm->reg_request->state = EC_INT_REQUEST_FAILURE;
369  wake_up_all(&slave->master->request_queue);
370  fsm->reg_request = NULL;
371  fsm->state = ec_fsm_slave_state_idle;
372  return 0;
373  }
374 
375  // Found pending register request. Execute it!
376  EC_SLAVE_DBG(slave, 1, "Processing register request...\n");
377 
378  fsm->reg_request->state = EC_INT_REQUEST_BUSY;
379 
380  // Start register access
381  if (fsm->reg_request->dir == EC_DIR_INPUT) {
382  ec_datagram_fprd(datagram, slave->station_address,
383  fsm->reg_request->address, fsm->reg_request->transfer_size);
384  ec_datagram_zero(datagram);
385  } else {
386  ec_datagram_fpwr(datagram, slave->station_address,
387  fsm->reg_request->address, fsm->reg_request->transfer_size);
388  memcpy(datagram->data, fsm->reg_request->data,
389  fsm->reg_request->transfer_size);
390  }
391  datagram->device_index = slave->device_index;
392  fsm->state = ec_fsm_slave_state_reg_request;
393  return 1;
394 }

400 void ec_fsm_slave_state_reg_request(
401  ec_fsm_slave_t *fsm,
402  ec_datagram_t *datagram
403  )
404 {
405  ec_slave_t *slave = fsm->slave;
406  ec_reg_request_t *reg = fsm->reg_request;
407 
408  if (!reg) {
409  // configuration was cleared in the meantime
410  fsm->state = ec_fsm_slave_state_ready;
411  fsm->reg_request = NULL;
412  return;
413  }
414 
415  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
416  EC_SLAVE_ERR(slave, "Failed to receive register"
417  " request datagram: ");
418  ec_datagram_print_state(fsm->datagram);
419  reg->state = EC_INT_REQUEST_FAILURE;
420  wake_up_all(&slave->master->request_queue);
421  fsm->reg_request = NULL;
422  fsm->state = ec_fsm_slave_state_ready;
423  return;
424  }
425 
426  if (fsm->datagram->working_counter == 1) {
427  if (reg->dir == EC_DIR_INPUT) { // read request
428  memcpy(reg->data, fsm->datagram->data, reg->transfer_size);
429  }
430 
431  reg->state = EC_INT_REQUEST_SUCCESS;
432  EC_SLAVE_DBG(slave, 1, "Register request successful.\n");
433  } else {
434  reg->state = EC_INT_REQUEST_FAILURE;
435  ec_datagram_print_state(fsm->datagram);
436  EC_SLAVE_ERR(slave, "Register request failed"
437  " (working counter is %u).\n",
438  fsm->datagram->working_counter);
439  }
440 
441  wake_up_all(&slave->master->request_queue);
442  fsm->reg_request = NULL;
443  fsm->state = ec_fsm_slave_state_ready;
444 }
