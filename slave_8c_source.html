176 void ec_slave_clear(ec_slave_t *slave )
177 {
178  ec_sdo_t *sdo, *next_sdo;
179  unsigned int i;
180  ec_pdo_t *pdo, *next_pdo;
181 
182  // abort all pending requests
183 
184  while (!list_empty(&slave->sdo_requests)) {
185  ec_sdo_request_t *request =
186  list_entry(slave->sdo_requests.next, ec_sdo_request_t, list);
187  list_del_init(&request->list); // dequeue
188  EC_SLAVE_WARN(slave, "Discarding SDO request,"
189  " slave about to be deleted.\n");
190  request->state = EC_INT_REQUEST_FAILURE;
191  }
192 
193  while (!list_empty(&slave->reg_requests)) {
194  ec_reg_request_t *reg =
195  list_entry(slave->reg_requests.next, ec_reg_request_t, list);
196  list_del_init(&reg->list); // dequeue
197  EC_SLAVE_WARN(slave, "Discarding register request,"
198  " slave about to be deleted.\n");
199  reg->state = EC_INT_REQUEST_FAILURE;
200  }
201 
202  while (!list_empty(&slave->foe_requests)) {
203  ec_foe_request_t *request =
204  list_entry(slave->foe_requests.next, ec_foe_request_t, list);
205  list_del_init(&request->list); // dequeue
206  EC_SLAVE_WARN(slave, "Discarding FoE request,"
207  " slave about to be deleted.\n");
208  request->state = EC_INT_REQUEST_FAILURE;
209  }
210 
211  while (!list_empty(&slave->soe_requests)) {
212  ec_soe_request_t *request =
213  list_entry(slave->soe_requests.next, ec_soe_request_t, list);
214  list_del_init(&request->list); // dequeue
215  EC_SLAVE_WARN(slave, "Discarding SoE request,"
216  " slave about to be deleted.\n");
217  request->state = EC_INT_REQUEST_FAILURE;
218  }
219 
220  while (!list_empty(&slave->eoe_requests)) {
221  ec_eoe_request_t *request =
222  list_entry(slave->eoe_requests.next, ec_eoe_request_t, list);
223  list_del_init(&request->list); // dequeue
224  EC_SLAVE_WARN(slave, "Discarding EoE request,"
225  " slave about to be deleted.\n");
226  request->state = EC_INT_REQUEST_FAILURE;
227  }
228 
229  wake_up_all(&slave->master->request_queue);
230 
231  if (slave->config) {
232  ec_slave_config_detach(slave->config);
233  }
234 
235  // free all SDOs
236  list_for_each_entry_safe(sdo, next_sdo, &slave->sdo_dictionary, list) {
237  list_del(&sdo->list);
238  ec_sdo_clear(sdo);
239  kfree(sdo);
240  }
241 
242  // free all strings
243  if (slave->sii.strings) {
244  for (i = 0; i < slave->sii.string_count; i++)
245  kfree(slave->sii.strings[i]);
246  kfree(slave->sii.strings);
247  }
248 
249  // free all sync managers
250  ec_slave_clear_sync_managers(slave);
251 
252  // free all SII PDOs
253  list_for_each_entry_safe(pdo, next_pdo, &slave->sii.pdos, list) {
254  list_del(&pdo->list);
255  ec_pdo_clear(pdo);
256  kfree(pdo);
257  }
258 
259  if (slave->sii_words) {
260  kfree(slave->sii_words);
261  }
262 
263  ec_fsm_slave_clear(&slave->fsm);
264 }

289 void ec_slave_set_dl_status(ec_slave_t *slave,
290  uint16_t new_state
291  )
292 {
293  unsigned int i;
294  uint8_t state;
295 
296  for (i = 0; i < EC_MAX_PORTS; i++) {
297  // link status
298  state = new_state & (1 << (4 + i)) ? 1 : 0;
299  if (slave->ports[i].link.link_up != state) {
300  EC_SLAVE_DBG(slave, 1, "Port %u link status changed to %s.\n",
301  i, state ? "up" : "down");
302  slave->ports[i].link.link_up = state;
303  }
304 
305  // loop status
306  state = new_state & (1 << (8 + i * 2)) ? 1 : 0;
307  if (slave->ports[i].link.loop_closed != state) {
308  EC_SLAVE_DBG(slave, 1, "Port %u loop status changed to %s.\n",
309  i, state ? "closed" : "open");
310  slave->ports[i].link.loop_closed = state;
311  }
312 
313  // signal detection
314  state = new_state & (1 << (9 + i * 2)) ? 1 : 0;
315  if (slave->ports[i].link.signal_detected != state) {
316  EC_SLAVE_DBG(slave, 1, "Port %u signal status changed to %s.\n",
317  i, state ? "yes" : "no");
318  slave->ports[i].link.signal_detected = state;
319  }
320  }
321 }

367 int ec_slave_fetch_sii_strings(
368  ec_slave_t *slave,
369  const uint8_t *data,
370  size_t data_size
371  )
372 {
373  int i, err;
374  size_t size;
375  off_t offset;
376 
377  slave->sii.string_count = data[0];
378 
379  if (slave->sii.string_count) {
380  if (!(slave->sii.strings =
381  kmalloc(sizeof(char *) * slave->sii.string_count,
382  GFP_KERNEL))) {
383  EC_SLAVE_ERR(slave, "Failed to allocate string array memory.\n");
384  err = -ENOMEM;
385  goto out_zero;
386  }
387 
388  offset = 1;
389  for (i = 0; i < slave->sii.string_count; i++) {
390  size = data[offset];
391  // allocate memory for string structure and data at a single blow
392  if (!(slave->sii.strings[i] =
393  kmalloc(sizeof(char) * size + 1, GFP_KERNEL))) {
394  EC_SLAVE_ERR(slave, "Failed to allocate string memory.\n");
395  err = -ENOMEM;
396  goto out_free;
397  }
398  memcpy(slave->sii.strings[i], data + offset + 1, size);
399  slave->sii.strings[i][size] = 0x00; // append binary zero
400  offset += 1 + size;
401  }
402  }
403 
404  return 0;
405 
406 out_free:
407  for (i--; i >= 0; i--)
408  kfree(slave->sii.strings[i]);
409  kfree(slave->sii.strings);
410  slave->sii.strings = NULL;
411 out_zero:
412  slave->sii.string_count = 0;
413  return err;
414 }

423 int ec_slave_fetch_sii_general(
424  ec_slave_t *slave,
425  const uint8_t *data,
426  size_t data_size
427  )
428 {
429  unsigned int i;
430  uint8_t flags;
431 
432  if (data_size != 32) {
433  EC_SLAVE_ERR(slave, "Wrong size of general category (%zu/32).\n",
434  data_size);
435  return -EINVAL;
436  }
437 
438  slave->sii.group = ec_slave_sii_string(slave, data[0]);
439  slave->sii.image = ec_slave_sii_string(slave, data[1]);
440  slave->sii.order = ec_slave_sii_string(slave, data[2]);
441  slave->sii.name = ec_slave_sii_string(slave, data[3]);
442 
443  for (i = 0; i < 4; i++)
444  slave->sii.physical_layer[i] =
445  (data[4] & (0x03 << (i * 2))) >> (i * 2);
446 
447  // read CoE details
448  flags = EC_READ_U8(data + 5);
449  slave->sii.coe_details.enable_sdo = (flags >> 0) & 0x01;
450  slave->sii.coe_details.enable_sdo_info = (flags >> 1) & 0x01;
451  slave->sii.coe_details.enable_pdo_assign = (flags >> 2) & 0x01;
452  slave->sii.coe_details.enable_pdo_configuration = (flags >> 3) & 0x01;
453  slave->sii.coe_details.enable_upload_at_startup = (flags >> 4) & 0x01;
454  slave->sii.coe_details.enable_sdo_complete_access = (flags >> 5) & 0x01;
455 
456  // read general flags
457  flags = EC_READ_U8(data + 0x000B);
458  slave->sii.general_flags.enable_safeop = (flags >> 0) & 0x01;
459  slave->sii.general_flags.enable_not_lrw = (flags >> 1) & 0x01;
460 
461  slave->sii.current_on_ebus = EC_READ_S16(data + 0x0C);
462  slave->sii.has_general = 1;
463  return 0;
464 }

474 int ec_slave_fetch_sii_syncs(
475  ec_slave_t *slave,
476  const uint8_t *data,
477  size_t data_size
478  )
479 {
480  unsigned int i, count, total_count;
481  ec_sync_t *sync;
482  size_t memsize;
483  ec_sync_t *syncs;
484  uint8_t index;
485 
486  // one sync manager struct is 4 words long
487  if (data_size % 8) {
488  EC_SLAVE_ERR(slave, "Invalid SII sync manager category size %zu.\n",
489  data_size);
490  return -EINVAL;
491  }
492 
493  count = data_size / 8;
494 
495  if (count) {
496  total_count = count + slave->sii.sync_count;
497  if (total_count > EC_MAX_SYNC_MANAGERS) {
498  EC_SLAVE_ERR(slave, "Exceeded maximum number of"
499  " sync managers!\n");
500  return -EOVERFLOW;
501  }
502  memsize = sizeof(ec_sync_t) * total_count;
503  if (!(syncs = kmalloc(memsize, GFP_KERNEL))) {
504  EC_SLAVE_ERR(slave, "Failed to allocate %zu bytes"
505  " for sync managers.\n", memsize);
506  return -ENOMEM;
507  }
508 
509  for (i = 0; i < slave->sii.sync_count; i++)
510  ec_sync_init_copy(syncs + i, slave->sii.syncs + i);
511 
512  // initialize new sync managers
513  for (i = 0; i < count; i++, data += 8) {
514  index = i + slave->sii.sync_count;
515  sync = &syncs[index];
516 
517  ec_sync_init(sync, slave);
518  sync->physical_start_address = EC_READ_U16(data);
519  sync->default_length = EC_READ_U16(data + 2);
520  sync->control_register = EC_READ_U8(data + 4);
521  sync->enable = EC_READ_U8(data + 6);
522  }
523 
524  if (slave->sii.syncs)
525  kfree(slave->sii.syncs);
526  slave->sii.syncs = syncs;
527  slave->sii.sync_count = total_count;
528  }
529 
530  return 0;
531 }

540 int ec_slave_fetch_sii_pdos(
541  ec_slave_t *slave,
542  const uint8_t *data,
543  size_t data_size,
544  ec_direction_t dir
545  )
546 {
547  int ret;
548  ec_pdo_t *pdo;
549  ec_pdo_entry_t *entry;
550  unsigned int entry_count, i;
551 
552  while (data_size >= 8) {
553  if (!(pdo = kmalloc(sizeof(ec_pdo_t), GFP_KERNEL))) {
554  EC_SLAVE_ERR(slave, "Failed to allocate PDO memory.\n");
555  return -ENOMEM;
556  }
557 
558  ec_pdo_init(pdo);
559  pdo->index = EC_READ_U16(data);
560  entry_count = EC_READ_U8(data + 2);
561  pdo->sync_index = EC_READ_U8(data + 3);
562  ret = ec_pdo_set_name(pdo,
563  ec_slave_sii_string(slave, EC_READ_U8(data + 5)));
564  if (ret) {
565  ec_pdo_clear(pdo);
566  kfree(pdo);
567  return ret;
568  }
569  list_add_tail(&pdo->list, &slave->sii.pdos);
570 
571  data_size -= 8;
572  data += 8;
573 
574  for (i = 0; i < entry_count; i++) {
575  if (!(entry = kmalloc(sizeof(ec_pdo_entry_t), GFP_KERNEL))) {
576  EC_SLAVE_ERR(slave, "Failed to allocate PDO entry memory.\n");
577  return -ENOMEM;
578  }
579 
580  ec_pdo_entry_init(entry);
581  entry->index = EC_READ_U16(data);
582  entry->subindex = EC_READ_U8(data + 2);
583  ret = ec_pdo_entry_set_name(entry,
584  ec_slave_sii_string(slave, EC_READ_U8(data + 3)));
585  if (ret) {
586  ec_pdo_entry_clear(entry);
587  kfree(entry);
588  return ret;
589  }
590  entry->bit_length = EC_READ_U8(data + 5);
591  list_add_tail(&entry->list, &pdo->entries);
592 
593  data_size -= 8;
594  data += 8;
595  }
596 
597  // if sync manager index is positive, the PDO is mapped by default
598  if (pdo->sync_index >= 0) {
599  ec_sync_t *sync;
600 
601  if (!(sync = ec_slave_get_sync(slave, pdo->sync_index))) {
602  EC_SLAVE_ERR(slave, "Invalid SM index %i for PDO 0x%04X.",
603  pdo->sync_index, pdo->index);
604  return -ENOENT;
605  }
606 
607  ret = ec_pdo_list_add_pdo_copy(&sync->pdos, pdo);
608  if (ret)
609  return ret;
610  }
611  }
612 
613  return 0;
614 }
